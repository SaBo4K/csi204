ผลการทดลอง Lab 1.2: เขียน Pseudocode

Pseudocode คำนวณพื้นที่วงกลม

// Pseudocode: การคำนวณพื้นที่วงกลม
BEGIN
    INPUT radius
    SET PI = 3.14159
    SET area = PI * radius * radius
    OUTPUT "Area of the circle is: " + area
END

Pseudocode คำนวณเกรด (A/B/C/D/F)

// Pseudocode: การคำนวณเกรดจากคะแนน 0-100
BEGIN
    INPUT score
    IF score >= 80 THEN
        SET grade = "A"
    ELSE IF score >= 70 THEN
        SET grade = "B"
    ELSE IF score >= 60 THEN
        SET grade = "C"
    ELSE IF score >= 50 THEN
        SET grade = "D"
    ELSE
        SET grade = "F"
    ENDIF
    OUTPUT "Your grade is: " + grade
END

Pseudocode หาค่าสูงสุดใน 3 จำนวน

// Pseudocode: การหาค่าสูงสุดจากตัวเลข 3 จำนวน
BEGIN
    INPUT num1, num2, num3
    SET max_value = num1
    
    IF num2 > max_value THEN
        SET max_value = num2
    ENDIF
    
    IF num3 > max_value THEN
        SET max_value = num3
    ENDIF
    
    OUTPUT "The maximum value is: " + max_value
END
